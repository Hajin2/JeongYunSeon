#include<stdio.h>
#include<conio.h>

void main()
{
	int Height;
	printf("세로 입력 : ");
	scanf("%d", &Height);
	for (int y = 0; y < Height; y++)
	{
		for (int x = 0; x < y; x++)
			printf("  ");
		int yMax = (Height - y) * 2 - 1;
		for (int x = 0; x < yMax; x++)
			printf("★");
		printf("\n");
	}
	getch();
}
